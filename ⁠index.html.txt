        const ISLAMIC_BLACKLIST = ['FUN', 'WIN', 'DYDX', 'GMX', 'GNS', 'PERP', 'HFT', 'AUCTION', 'JOE'];
        const MIN_VOLUME_USD = 20000000; // 20 مليون دولار سيولة كما اقترح الخبير لفتح نطاق الفرص

            const grid = document.getElementById('grid');
            const statusText = document.getElementById('statusText');
            const scannedCountText = document.getElementById('scannedCount');

                // 1️⃣ اختبار فلتر البيتكوين الحاكم أولاً لحماية الحساب (15m و 1h)

                const btc15 = await btc15Resp.json();
                const btcCloses = btc15.map(k => parseFloat(k[4]));

                    statusText.innerText = "🚨 حظر اتجاه البيتكوين!";
                    grid.innerHTML = '<div class="no-data" style="color:var(--danger);">⚠️ الرادار متوقف تلقائياً: اتجاه البيتكوين سلبي أو هابط (السعر تحت الـ EMA50 أو الماكد سلبي). الكاش هو الملك الآن لحماية رأس مالك!</div>';
                    scanBtn.disabled = false;
                    return;
                }

                const qualifiedSymbols = allTickers.filter(t => {
                    return t.symbol.endsWith('USDT') && 
                           t.symbol !== 'BTCUSDT' &&
                           !ISLAMIC_BLACKLIST.includes(t.symbol.replace('USDT', '')) && 
                           parseFloat(t.quoteVolume) >= MIN_VOLUME_USD &&
                           parseInt(t.count) >= MIN_TRADES_COUNT;
                });

                // ترتيب كامل العملات المؤهلة حسب السيولة واستهداف فحص أعلى 40 عملة حية حركياً لمنع ثقل المتصفح
                const targetPairs = qualifiedSymbols.sort((a,b) => parseFloat(b.quoteVolume) - parseFloat(a.quoteVolume)).slice(0, 40);
                let foundOpportunities = [];

                for(let i=0; i<targetPairs.length; i++) {
                    const coin = targetPairs[i];
                    const coinSymbol = coin.symbol.replace('USDT', '');
                    statusText.innerText = `تحليل كمّي عميق (${i+1}/${targetPairs.length})`;

                    try {
                        // طلب واحد فقط لكل عملة يحتوي على 200 شمعة (توفير 300% من حجم الطلبات السابقة)
                        const kResp = await fetch(`https://api.binance.com/api/v3/klines?symbol=${coin.symbol}&interval=15m&limit=200`);
                        const klines = await kResp.json();
                        if(klines.length < 150) continue;

                        const closes = klines.map(k => parseFloat(k[4]));
                        const highs = klines.map(k => parseFloat(k[2]));
                        const lows = klines.map(k => parseFloat(k[3]));
                        const volumes = klines.map(k => parseFloat(k[5]));

                        const rsi = calculateWilderRSI(closes, 14);

                        const macd = calculateTrueMACD(closes);

                        // 📈 حساب الفلاتر الاحترافية الجديدة الموصى بها
                        // أ- حساب حجم التداول النسبي RVOL لآخر 50 شمعة
                        const avgVolume50 = volumes.slice(-51, -1).reduce((a,b)=>a+b, 0) / 50;
                        const rvol = volumes[volumes.length - 1] / avgVolume50;

                        // ب- حساب انفجار التذبذب ATR Expansion
                        const avgAtrPrior = calculateWilderATR(highs.slice(0, -1), lows.slice(0, -1), closes.slice(0, -1), 14);
                        const atrExpansion = atr > avgAtrPrior;

                        // 🎯 معادلة معايرة توزيع الأوزان النسبية الصارمة للمستشار (إجمالي 100)
                        let totalScore = 0;
                        let triggers = [];

                        if(ema20 > ema50) { totalScore += 25; triggers.push("بنية اتجاهية صاعدة (EMA)"); }
                        if(currentPrice > Math.max(...highs.slice(-21, -1)) && rvol > 1.8) { totalScore += 25; triggers.push("اختراق قمة مؤكد بـ RVOL"); }
                        if(macd.histogram > 0 && macd.macdLine > macd.signalLine) { totalScore += 20; triggers.push("تقاطع MACD إيجابي"); }
                        if(rsi >= 55 && rsi <= 70) { totalScore += 15; triggers.push("زخم RSI مثالي"); }
                        if(atrExpansion) { totalScore += 10; triggers.push("انفجار تذبذب سيولة (ATR Exp)"); }

                        // فرز الفرص الذهبية الحقيقية فقط (الحد الأدنى 65 نقطة لضمان القوة)
                        if(totalScore >= 65) {
                            const stopLoss = currentPrice - (1.0 * atr); // وقف خسارة صارم 1x ATR لحماية الحساب
                            const targetProfit = currentPrice * 1.02; // هدف الاستراتيجية الثابت 2%

                                <div class="metric-item"><span class="label">الفلترة الشرعية المحددة</span><span class="val" style="color:var(--info); font-size:0.65rem;">لم يقع في قائمة الحظر المحددة</span></div>
                            </div>

                                <span style="color:var(--primary); font-weight:bold; display:block; margin-bottom:4px;">⚔️ الأوامر التنفيذية لـ استراتيجية الـ 2%:</span>
                                • نقطة الدخول: <b>${item.entry}</b><br>
                                • هدف جنى الأرباح المأمون (+2%): <b style="color:var(--primary);">${item.tp.toFixed(5)}</b><br>
                                • وقف الخسارة الصارم (1x Wilder ATR): <b style="color:var(--danger);">${item.sl.toFixed(5)}</b><br>
                                <span style="color:var(--text-muted); font-size:0.7rem; display:block; margin-top:5px;">💡 مبررات الخوارزمية الفنية: ${item.triggers.join(' | ')}</span>
                            </div>

                statusText.innerText = "خطأ اتصال";
                grid.innerHTML = '<div class="no-data">❌ فشل جلب البيانات، يرجى المحاولة لاحقاً.</div>';

        }

        function calculateTrueMACD(closes) {
            let ema12List = [], ema26List = [], macdLines = [];
            let ema12 = closes[0], ema26 = closes[0];
            for(let i=0; i<closes.length; i++) {
                ema12 = (closes[i] * (2/13)) + (ema12 * (1 - (2/13)));
                ema26 = (closes[i] * (2/27)) + (ema26 * (1 - (2/27)));
                macdLines.push(ema12 - ema26);
            }
            let signalLine = macdLines[0];
            for(let j=1; j<macdLines.length; j++) { signalLine = (macdLines[j] * (2/10)) + (signalLine * (1 - (2/10))); }
            const currentMacd = macdLines[macdLines.length - 1];
            return { macdLine: currentMacd, signalLine: signalLine, histogram: currentMacd - signalLine };
        }

        function calculateWilderATR(highs, lows, closes, period = 14) {
            let trs = [];
            for(let i=1; i<closes.length; i++) {
                let tr = Math.max(highs[i] - lows[i], Math.abs(highs[i] - closes[i-1]), Math.abs(lows[i] - closes[i-1]));
                trs.push(tr);
            }
            let atr = trs.slice(0, period).reduce((a,b)=>a+b, 0) / period;
            for(let i=period; i<trs.length; i++) { atr = ((atr * (period - 1)) + trs[i]) / period; }
            return atr;
        }

        function calculateWilderRSI(closes, period = 14) {
            let changes = [];
            for(let i=1; i<closes.length; i++) changes.push(closes[i] - closes[i-1]);
            let gains = changes.map(c => c > 0 ? c : 0);
            let losses = changes.map(c => c < 0 ? -c : 0);
            let avgGain = gains.slice(0, period).reduce((a,b)=>a+b, 0) / period;
            let avgLoss = losses.slice(0, period).reduce((a,b)=>a+b, 0) / period;
            for(let i=period; i<changes.length; i++) {
                avgGain = ((avgGain * (period - 1)) + gains[i]) / period;
                avgLoss = ((avgLoss * (period - 1)) + losses[i]) / period;
            }
            if(avgLoss === 0) return 100;
            return 100 - (100 / (1 + (avgGain / avgLoss)));
        }

        function calculateEMA(closes, period) {
            const k = 2 / (period + 1); let ema = closes[0];
            for (let i = 1; i < closes.length; i++) { ema = (closes[i] * k) + (ema * (1 - k)); }
            return ema;
        }

        function saveSignalToJournalV24(symbol, entry, tp, sl) {
            let journal = JSON.parse(localStorage.getItem('v24_journal')) || [];
            if(journal.some(j => j.symbol === symbol && j.status === 'PENDING')) return;
            journal.push({ symbol, entry, tp, sl, status: 'PENDING', time: Date.now() });
            if(journal.length > 300) journal.shift();
            localStorage.setItem('v24_journal', JSON.stringify(journal));
        }

        // 📊 دالة محاكاة التنفيذ الحقيقي المتقدمة المعتمدة على فحص الشموع الكاملة والـ High / Low
        async function updateJournalResultsV24() {
            let journal = JSON.parse(localStorage.getItem('v24_journal')) || [];
            let pendings = journal.filter(j => j.status === 'PENDING');
            if(pendings.length === 0) { updateJournalSummaryDOMV24(); return; }

            for (let job of journal) {
                if(job.status === 'PENDING') {
                    try {
                        // سحب شمعة حية للتحقق من أعلى وأدنى سعر حقيقي تم تسجيله منذ الإشارة
                        const res = await fetch(`https://api.binance.com/api/v3/klines?symbol=${job.symbol}USDT&interval=15m&limit=12`);
                        const candles = await res.json();
                        
                        let hitWin = false;
                        let hitLoss = false;

                        for(let c of candles) {
                            const candleHigh = parseFloat(c[2]);
                            const candleLow = parseFloat(c[3]);

                            if(candleHigh >= job.tp) hitWin = true;
                            if(candleLow <= job.sl) hitLoss = true;
                        }

                        if(hitWin) job.status = 'WIN';
                        else if(hitLoss) job.status = 'LOSS';
                        else if(Date.now() - job.time > 7200000) job.status = 'EXPIRED'; // فصل المنتهية الصلاحية تماماً لحفظ دقة الإحصاء

                    } catch(e) { console.log('تخطي تحديث العملة الفردية عابراً'); }
                }
            }
            localStorage.setItem('v24_journal', JSON.stringify(journal));
            updateJournalSummaryDOMV24();
        }

        function updateJournalSummaryDOMV24() {
            const summaryDiv = document.getElementById('journalSummary');
            let journal = JSON.parse(localStorage.getItem('v24_journal')) || [];
            let finished = journal.filter(j => j.status !== 'PENDING');
            if(finished.length === 0) {
                summaryDiv.innerHTML = "📊 <b>المختبر الإحصائي V24.0:</b> لا توجد صفقات منتهية بعد. بادر بالمسح لتجميع بيانات التداول والـ Backtesting.";
                return;
            }
            let wins = finished.filter(j => j.status === 'WIN').length;
            let losses = finished.filter(j => j.status === 'LOSS').length;
            let expired = finished.filter(j => j.status === 'EXPIRED').length;
            let totalWinRate = (wins / finished.length * 100).toFixed(1);
            
            summaryDiv.innerHTML = `📊 <b>بيانات القياس والأداء الفعلي الحقيقي (Backtesting):</b> تم تتبع <b>${finished.length}</b> صفقة من كل السوق.<br>• نسبة النجاح الحقيقية: <span style="color:var(--primary); font-weight:bold;">${totalWinRate}%</span> | صفقات رابحة: <span style="color:var(--primary);">${wins}</span> | صفقات خاسرة: <span style="color:var(--danger);">${losses}</span> | صفقات منتهية الوقت (Expired): <span style="color:var(--warning);">${expired}</span>`;
        }
